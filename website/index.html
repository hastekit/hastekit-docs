main.go 
package main

import (
    "os"
    hastekit "github.com/hastekit/hastekit-sdk-go"
    "github.com/hastekit/hastekit-sdk-go/pkg/gateway"
    "github.com/hastekit/hastekit-sdk-go/pkg/gateway/llm"
)

func main() {
    // Initialize the SDK with provider configuration
    client, err := hastekit.New(&hastekit.ClientOptions{
        ProviderConfigs: []gateway.ProviderConfig{
            {
                ProviderName: llm.ProviderNameOpenAI,
                ApiKeys: []*gateway.APIKeyConfig{
                    {
                        Name:   "Key 1",
                        APIKey: os.Getenv("OPENAI_API_KEY"),
                    },
                },
            },
        },
    })
}
import "github.com/hastekit/hastekit-sdk-go/pkg/gateway/llm/responses"

// Make LLM calls with the Responses API
resp, err := client.NewResponses(ctx, &responses.Request{
    Model:        "OpenAI/gpt-4.1-mini",
    Instructions: utils.Ptr("You are a helpful assistant."),
    Input: responses.InputUnion{
        OfString: utils.Ptr("What is the capital of France?"),
    },
})

// Extract the generated text from the response
for _, output := range resp.Output {
    if output.OfOutputMessage != nil {
        for _, content := range output.OfOutputMessage.Content {
            if content.OfOutputText != nil {
                fmt.Println(content.OfOutputText.Text)
            }
        }
    }
}
// Create an agent with system instructions
agent := client.NewAgent(&hastekit.AgentOptions{
    Name:        "Assistant",
    Instruction: client.Prompt("You are a helpful assistant."),
    LLM: client.NewLLM(hastekit.LLMOptions{
        Provider: llm.ProviderNameOpenAI,
        Model:    "gpt-4o-mini",
    }),
})

// Execute the agent with user messages
out, err := agent.Execute(ctx, &agents.AgentInput{
    Messages: []responses.InputMessageUnion{
        responses.UserMessage("Hello!"),
    },
})

// Extract the agent's response
for _, output := range out.Output {
    if output.OfOutputMessage != nil {
        for _, content := range output.OfOutputMessage.Content {
            if content.OfOutputText != nil {
                fmt.Println(content.OfOutputText.Text)
            }
        }
    }
}

LLM Gateway

Unified Access to All LLM Providers

Route requests to any LLM provider through a single, secure gateway. No more managing multiple API keys in your applications.

  • Provider Abstraction — Access OpenAI, Anthropic, Gemini, xAI, and Ollama through unified endpoints
  • Virtual Keys — Generate sk-hk-* keys with fine-grained access control and rate limits
  • Drop-in Replacement — Works with existing OpenAI, Anthropic, and Gemini SDKs
  • Deep Observability — OpenTelemetry traces, token usage, latency metrics via ClickHouse
main.py 
from openai import OpenAI

# Point to HasteKit Gateway instead of OpenAI directly
client = OpenAI(
    base_url="http://localhost:6060/api/gateway/openai",
    api_key="your-virtual-key--or--provider-api-key",
)

# Use the Responses API
response = client.responses.create(
    model="gpt-4.1-mini",
    instructions="You are a coding assistant that talks like a pirate.",
    input="Hello!",
)

print(response.output_text)
